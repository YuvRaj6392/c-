#include<stdio.h>
#include<stdlib.h>
int main()
{
	int row1,row2,col1,col2,sum=0;
printf("Enter the no. of rows in a\n");
scanf("%d",&row1);
printf("Enter the no. of cols in a\n");
scanf("%d",&col1);
printf("Enter the no. of rows in b\n");
scanf("%d",&row2);
printf("Enter the no. of cols in b\n");
scanf("%d",&col2);
int a[row1][col1],b[row2][col2],c[row2][col1];
if(col1!=row1)
{
printf("Calculation not possible\n");
}
else
{
printf("Enter elements of a\n");
for(int i=0;i<row1;i++)
{
for(int j=0;j<col1;j++)
{
scanf("%d",&a[i][j]);
}
}
printf("Enter elements of b\n");
for(int i=0;i<row2;i++)
{
for(int j=0;j<col2;j++)
{
scanf("%d",&b[i][j]);
}
}
printf("Elements of a are\n");
for(int i=0;i<row1;i++)
{
for(int j=0;j<col1;j++)
{
printf("%d ",a[i][j]);
}
printf("\n");
}

printf("Elements of b are\n");
for(int i=0;i<row2;i++)
{
for(int j=0;j<col2;j++)
{
printf("%d ",b[i][j]);
}
printf("\n");
}
for(int i=0;i<row1;i++)
{
for(int j=0;j<col1;j++)
{
for(int m=0;m<col2;m++)
	{
		sum+=a[i][j]*b[m][j];
		c[i][j]=sum;
	}
sum=0;
}
}
}
printf("Elements of c are\n");
for(int i=0;i<row2;i++)
{
for(int j=0;j<col1;j++)
{
printf("%d ",c[i][j]);
}
printf("\n");
}
return 0;
}
