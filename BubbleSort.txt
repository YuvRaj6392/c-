#include<stdio.h>
#include<stdlib.h>
void printarray(int a[],int n)
{
	for(int i=0;i<n;i++)
	{
		printf("%d ",a[i]);
	}
	printf("\n");
}
void bubble(int a[],int n)
{
	int temp=0;
	int sort=0;
	for(int i=0;i<n-1;i++)
	{
		printf("on phase %d\n",i+1);
		sort=1;
		for(int j=0;j<n-1-i;j++)
		{
			if(a[j]>a[j+1])
			{
				temp=a[j];
				a[j]=a[j+1];
				a[j+1]=temp;
				sort=0;
			}
		}
		if(sort)
		{
			return;
		}
	}
}
int main()
{
int a[6]={6,5,1,2,4,3};
int n=6;
printarray(a,n);
bubble(a,n);
printarray(a,n);
return 0;
}
