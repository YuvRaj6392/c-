#include <stdio.h>
#include <stdlib.h>

struct stac
{
    int size;
    int top;
    char *arr;
};

int isEmpty(struct stac *ptr)
{
    if (ptr->top == -1)
    {
        return 1;
    }
    else
    {
        return 0;
    }
}

int isFull(struct stac *ptr)
{
    if (ptr->top == ptr->size - 1)
    {
        return 1;
    }
    else
    {
        return 0;
    }
}

void push(struct stac *ptr, char val){
    if(isFull(ptr)){
        printf("Stack Overflow! Cannot push %d to the stack\n", val);
    }
    else{
        ptr->top++;
        ptr->arr[ptr->top] = val;
    }
}

char pop(struct stac *ptr){
    if(isEmpty(ptr)){
        printf("Stack Underflow! Cannot pop from the stack\n");
        return -1;
    }
    else{
        char val = ptr->arr[ptr->top];
        ptr->top--;
        return val;
    }
}

int parenthesisMatch(char *exp){
    // Create and initialize the stack
    struct stac *sp;
    sp=(struct stac*)malloc(sizeof(struct stac));
    sp->size = 10;
    sp->top = -1;
    sp->arr = (char *)malloc(sp->size * sizeof(char));


    for (int i = 0; exp[i]!='\0'; i++)
    {
        if(exp[i]=='(' || exp[i]=='[' || exp[i]=='{'){
            push(sp,exp[i]);
        }
        else if(exp[i]==')')
        {
        	if(isEmpty(sp))
        		{
        			return 0;
        		}
        	else
        	{
        		if(sp->arr[sp->top]=='(')
        			{
        				pop(sp);
        			}
        			else
        			{
        				return 0;
        			}
        	}
        }
        
        
        else if(exp[i]==']')
        {
        	if(isEmpty(sp))
        		{
        			return 0;
        		}
        	else
        	{
        		if(sp->arr[sp->top]=='[')
        			{
        				pop(sp);
        			}
        			else
        			{
        				return 0;
        			}
        	}
        }
        
        
        else if(exp[i]=='}')
        {
        	if(isEmpty(sp))
        		{
        			return 0;
        		}
        	else
        	{
        		if(sp->arr[sp->top]=='{')
        			{
        				pop(sp);
        			}
        			else
        			{
        				return 0;
        			}
        	}
        }
        
    }

    if(isEmpty(sp)){
        return 1;
    }
    else{
        return 0;
    }
    
}
int main()
{
    char *exp = "{7-(3-2)+[8+(9-11)]}()(";
    // Check if stack is empty
    if(parenthesisMatch(exp)){
        printf("The parenthesis is matching");
    }
    else{
        printf("The parenthesis is not matching");
    }
    return 0;
}
