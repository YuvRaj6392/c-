void enque(int val)
{
	struct node *n=(struct node *)malloc(sizeof(struct node));
	n->data=val;
	n->next=NULL;
	if(n==NULL)
	{
		printf("Sorry, you cannot insert more!");
	}
	else
	{
		if(f==NULL)
		{
			f=r=n;
			printf("There was a successfull insertion\n");
		}
		else
		{
			r->next=n;
			r=r->next;
			printf("There was a successfull insertion\n");
		}
	}
}
