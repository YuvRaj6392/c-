#include<stdio.h>
#include<stdlib.h>
struct myarray
{
int t_size;
int u_size;
int* ptr;
};
void createarray(struct myarray* a,int t,int u)
{
a->t_size=t;
a->u_size=u;
a->ptr=(int*)malloc(t*sizeof(int));
}
void setarray(struct myarray* a)
{
int n;
for(int i=0;i< a->u_size;i++)
{
scanf("%d",&n);
(a->ptr[i])=n;
}
}
void showarray(struct myarray* a)
{
for(int i=0;i< a->u_size;i++)
{
printf("%d ",(a->ptr)[i]);
}
}
int main()
{
struct myarray marks;
createarray(&marks,100,5);
setarray(&marks);
showarray(&marks);
return 0;
}
